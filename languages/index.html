Python - example.py

#!/usr/bin/python
import random
# Used for a python rock-paper-scissors game
def get_best_move():
	global rock_number
	global paper_number
	global scissors_number

	if rock_number > paper_number and rock_number > scissors_number:
		return "p"
	elif paper_number > rock_number and paper_number > scissors_number:
		return "s"
	elif scissors_number > rock_number and scissors_number > paper_number:
		return "r"
	else:
		return random.choice(["r", "p", "s"])
